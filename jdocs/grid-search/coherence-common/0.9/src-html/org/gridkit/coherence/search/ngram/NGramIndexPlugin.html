017    package org.gridkit.coherence.search.ngram;
018    
019    import java.util.HashSet;
020    import java.util.Map;
021    import java.util.Set;
022    
023    import org.gridkit.coherence.search.IndexEngineConfig;
024    import org.gridkit.coherence.search.IndexInvocationContext;
025    import org.gridkit.coherence.search.IndexUpdateEvent;
026    import org.gridkit.coherence.search.PlugableSearchIndex;
027    
028    /**
029     * Reference implementation of custom index.
030     * 
031     * @author Alexey Ragozin (alexey.ragozin@gmail.com)
032     */
033    public class NGramIndexPlugin implements PlugableSearchIndex<NGramIndex, Integer, String> {
034    
035            @Override
036            public boolean applyIndex(NGramIndex index, String query, Set<Object> keySet, IndexInvocationContext context) {
037            return ngramFilter(index, query, keySet);
038        }
039    
040        @SuppressWarnings("unchecked")
041            private boolean ngramFilter(NGramIndex index, String substring, Set setKeys) {
042            if (index.getNGramSize() <= substring.length()) {
043                filterSet(index, substring, setKeys);
044                return true;
045            }
046            else {
047                unionFilterSet(index, substring, setKeys);
048                return false;
049            }
050        }
051    
052        @SuppressWarnings("unchecked")
053            private void unionFilterSet(NGramIndex index, String substring, Set setKeys) {
054            Set result = new HashSet();
055            for(Object entry: index.getNGramMap().entrySet()) {
056                String ngram = (String) ((Map.Entry)entry).getKey();
057                NGramRefList list = (NGramRefList) ((Map.Entry)entry).getValue();
058                
059                if (ngram.indexOf(substring) >= 0) {
060                    for(Object key: list.references.keySet()) {
061                        if (setKeys.contains(key)) {
062                            result.add(key);
063                        }
064                    }
065                }            
066            }
067            setKeys.retainAll(result);
068        }
069    
070        @SuppressWarnings("unchecked")
071            private void filterSet(NGramIndex index, String substring, Set setKeys) {
072            int n = 0;
073            int ngramSize = index.getNGramSize();
074            while(true) {
075                String ngram = substring.substring(n, n + ngramSize);
076                setKeys.retainAll(index.lookUpNGram(ngram));     
077                if (setKeys.isEmpty()) {
078                    break;
079                }
080                n += ngramSize;
081                if (n == substring.length()) {
082                    break;
083                }
084                else if (n + ngramSize > substring.length()) {
085                    int over = n + ngramSize - substring.length();
086                    n -= over;
087                }
088            }
089        }
090    
091            @Override
092            public int calculateEffectiveness(NGramIndex index, String query, Set<Object> keySet, IndexInvocationContext context) {
093                    if (query.length() > index.getNGramSize()) {
094                            return query.length() / index.getNGramSize() + 1;
095                    }
096                    else {
097                            // TODO
098                            return 2 * index.getNGramMap().size();
099                    }
100            }
101    
102            @Override
103            public void configure(IndexEngineConfig config) {
104                    // use defaults
105            }
106    
107            @Override
108            public Object createIndexCompatibilityToken(Integer indexConfig) {
109                    return NGramIndexToken.INSTANCE;
110            }
111    
112            @Override
113            public NGramIndex createIndexInstance(Integer indexConfig) {
114                    return new NGramIndex(indexConfig);
115            }
116    
117            @Override
118            public boolean evaluate(String query, Object document) {
119                    String text = String.valueOf(document);
120                    return text.contains(query);
121            }
122    
123            @Override
124            public void updateIndexEntries(NGramIndex index, Map<Object, IndexUpdateEvent> events, IndexInvocationContext context) {
125                    for (IndexUpdateEvent event: events.values()) {
126                            switch (event.getType()) {
127                            case INSERT:
128                                    index.insert(event.getKey(), String.valueOf(event.getValue()));
129                                    break;
130                            case UPDATE:
131                                    index.update(event.getKey(), String.valueOf(event.getValue()));
132                                    break;
133                            case DELETE:
134                                    index.delete(event.getKey());
135                                    break;
136                            }
137                    }
138            }
139    }
