017    package org.gridkit.coherence.search;
018    
019    import java.io.Serializable;
020    
021    /**
022     * Default implementation of {@link IndexEngineConfig}. 
023     * @author Alexey Ragozin (alexey.ragozin@gmail.com)
024     */
025    public class DefaultIndexEngineConfig implements IndexEngineConfig, Serializable {
026    
027            private int indexUpdateQueueSizeLimit = 1024;
028            private int indexUpdateDelay = 5000;
029            private boolean attributeIndexEnabled = false;
030            private boolean oldValueOnUpdateEnabled = false;
031            
032            public int getIndexUpdateQueueSizeLimit() {
033                    return indexUpdateQueueSizeLimit;
034            }
035            
036            public void setIndexUpdateQueueSizeLimit(int indexUpdateQueueSizeLimit) {
037                    this.indexUpdateQueueSizeLimit = indexUpdateQueueSizeLimit;
038            }
039            
040            public int getIndexUpdateDelay() {
041                    return indexUpdateDelay;
042            }
043            
044            public void setIndexUpdateDelay(int indexUpdateDelay) {
045                    this.indexUpdateDelay = indexUpdateDelay;
046            }
047            
048            public boolean isAttributeIndexEnabled() {
049                    return attributeIndexEnabled;
050            }
051            
052            public void setAttributeIndexEnabled(boolean attributeIndexEnabled) {
053                    this.attributeIndexEnabled = attributeIndexEnabled;
054            }
055            
056            public boolean isOldValueOnUpdateEnabled() {
057                    return oldValueOnUpdateEnabled;
058            }
059    
060            public void setOldValueOnUpdateEnabled(boolean oldValueOnUpdateEnabled) {
061                    this.oldValueOnUpdateEnabled = oldValueOnUpdateEnabled;
062            }
063    }
